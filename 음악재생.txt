import os
import glob

import subprocess

def happy_mp3():
	randomfile = random.choice(os.listdir("/home/pi/Documents/Python_projects/happy_mp3/"))
	print(randomfile)
	speak = "mpg321 ./happy_mp3/"+ randomfile +" -a convertQBO"
	result = subprocess.call(speak, shell = True)
